#----------------------------------------------------------------------------
# Setup the project
cmake_minimum_required(VERSION 2.6 FATAL_ERROR)
project(VoxelizedHumanDoseMultiSDv3)

SET(CMAKE_MODULE_PATH /usr/local/geant4/geant4.9.6.p02-install/lib/Geant4-9.6.2/Modules)

#----------------------------------------------------------------------------
# Find Geant4 package, activating all available UI and Vis drivers by default
# You can set WITH_GEANT4_UIVIS to OFF via the command line or ccmake/cmake-gui
# to build a batch mode only executable
#
option(WITH_GEANT4_UIVIS "Build example with Geant4 UI and Vis drivers" ON)
if(WITH_GEANT4_UIVIS)
  find_package(Geant4 REQUIRED ui_all vis_all)
else()
  find_package(Geant4 REQUIRED)
endif()


#----------------------------------------------------------------------------
# Find ROOT variables if the variable G4ANALYSIS_USE_ROOT is set
#
find_package(ROOT)


#----------------------------------------------------------------------------
# Setup Geant4 include directories and compile definitions
# Setup include directory for this project
#
if(ROOT_FOUND)
add_definitions(-DG4ANALYSIS_USE_ROOT)
include(${Geant4_USE_FILE})
include_directories(${ROOT_INCLUDE_DIR} ${Geant4_INCLUDE_DIR} ${PROJECT_SOURCE_DIR}/include)
else()
include(${Geant4_USE_FILE})
include_directories(${PROJECT_SOURCE_DIR}/include)
endif()


#----------------------------------------------------------------------------
# Locate sources and headers for this project
# NB: headers are included so they will show up in IDEs
#
file(GLOB sources ${PROJECT_SOURCE_DIR}/src/*.cc)
file(GLOB headers ${PROJECT_SOURCE_DIR}/include/*.hh)

#----------------------------------------------------------------------------
# Add the executable, and link it to the Geant4 libraries
#
add_executable(VHDMSDv3 VoxelizedHumanDoseMultiSDv3.cc ${sources} ${headers})

if(ROOT_FOUND)
target_link_libraries(VHDMSDv3 ${Geant4_LIBRARIES} ${ROOT_LIBRARIES})
else()
target_link_libraries(VHDMSDv3 ${Geant4_LIBRARIES})
endif()


#----------------------------------------------------------------------------
# Copy all scripts to the build directory, i.e. the directory in which we
# build B1. This is so that we can run the executable directly because it
# relies on these scripts being in the current working directory.
#

set(EXAMPLEVHDMSDv3_SCRIPTS
#  clean.sh
  macro/vis.mac
  macro/I131_Emphysics1.mac
  macro/I131_Emphysics2.mac
  runscript/VHDMSDv3MultiRun.sh
  runscript/multijob.sh
  runscript/VHDMSDv3MultiRun3.sh
  runscript/sleep.sh
  runscript/MIBGPT3.sh
  rootC/Root2Dat_EdepTree.C
  rootC/Root2Dat_FluenceTree.C
  rootC/Root2Dat_SrcEngHIST.C
  rootC/Root2Dat_FluenceAllTree.C
)

foreach(_script ${EXAMPLEVHDMSDv3_SCRIPTS})
  configure_file(
    ${PROJECT_SOURCE_DIR}/${_script}
    ${PROJECT_BINARY_DIR}/${_script}
    COPYONLY
    )
endforeach()

#----------------------------------------------------------------------------
# For internal Geant4 use - but has no effect if you build this
# example standalone
#
add_custom_target(VoxelizedHumanDoseMultiSDv3 DEPENDS VHDMSDv3)

#----------------------------------------------------------------------------
# Install the executable to 'bin' directory under CMAKE_INSTALL_PREFIX
#
install(TARGETS VHDMSDv3 DESTINATION bin)
